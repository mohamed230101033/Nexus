Basic String Encryption

# String Obfuscation Example (Educational)
import base64
import codecs

class StringObfuscator:
    def __init__(self):
        self.key = b'education_key_123'
    
    def xor_encrypt(self, data, key):
        """Simple XOR encryption for string obfuscation"""
        result = []
        for i, byte in enumerate(data):
            if isinstance(byte, str):
                byte = ord(byte)
            key_byte = key[i % len(key)]
            if isinstance(key_byte, str):
                key_byte = ord(key_byte)
            result.append(byte ^ key_byte)
        return bytes(result)
    
    def obfuscate_string(self, plaintext):
        """Multi-layer string obfuscation"""
        # Layer 1: XOR encryption
        encrypted = self.xor_encrypt(plaintext.encode(), self.key)
        
        # Layer 2: Base64 encoding
        b64_encoded = base64.b64encode(encrypted)
        
        # Layer 3: ROT13 on the base64
        rot13_encoded = codecs.encode(b64_encoded.decode(), 'rot13')
        
        return rot13_encoded
    
    def deobfuscate_string(self, obfuscated):
        """Reverse the obfuscation process"""
        try:
            # Reverse Layer 3: ROT13 decode
            rot13_decoded = codecs.decode(obfuscated, 'rot13')
            
            # Reverse Layer 2: Base64 decode
            b64_decoded = base64.b64decode(rot13_decoded.encode())
            
            # Reverse Layer 1: XOR decrypt
            decrypted = self.xor_encrypt(b64_decoded, self.key)
            
            return decrypted.decode()
        except Exception as e:
            return f"Deobfuscation failed: {e}"

# Example usage for educational analysis
obfuscator = StringObfuscator()
original = "This is a test string"
obfuscated = obfuscator.obfuscate_string(original)
print(f"Original: {original}")
print(f"Obfuscated: {obfuscated}")
print(f"Deobfuscated: {obfuscator.deobfuscate_string(obfuscated)}")

# Email Header Analysis Tool (Educational)
import email
import re
from datetime import datetime

class PhishingAnalyzer:
    def __init__(self):
        self.suspicious_domains = [
            'tempmail.org', '10minutemail.com', 'guerrillamail.com',
            'mailinator.com', 'throwaway.email'
        ]
        self.phishing_indicators = [
            'urgent', 'immediate action', 'verify account',
            'suspended', 'click here now', 'limited time'
        ]
    
    def analyze_headers(self, email_content):
        """Analyze email headers for suspicious indicators"""
        msg = email.message_from_string(email_content)
        analysis = {
            'sender_reputation': self.check_sender_reputation(msg),
            'spf_check': self.check_spf(msg),
            'dkim_verification': self.check_dkim(msg),
            'suspicious_headers': self.detect_suspicious_headers(msg)
        }
        return analysis
    
    def check_sender_reputation(self, msg):
        """Check if sender domain is suspicious"""
        sender = msg.get('From', '')
        domain_match = re.search(r'@([a-zA-Z0-9.-]+)', sender)
        
        if domain_match:
            domain = domain_match.group(1).lower()
            if domain in self.suspicious_domains:
                return {'status': 'suspicious', 'reason': 'Known disposable email'}
            
            # Check for domain spoofing attempts
            if self.check_domain_spoofing(domain):
                return {'status': 'suspicious', 'reason': 'Potential domain spoofing'}
        
        return {'status': 'clean', 'reason': 'No immediate red flags'}
    
    def check_domain_spoofing(self, domain):
        """Detect potential domain spoofing"""
        legitimate_domains = ['google.com', 'microsoft.com', 'amazon.com']
        
        for legit_domain in legitimate_domains:
            # Check for character substitution
            if self.similar_domain(domain, legit_domain):
                return True
        return False
    
    def similar_domain(self, domain1, domain2):
        """Check if domains are suspiciously similar"""
        # Simple similarity check (educational purposes)
        if len(domain1) == len(domain2):
            diff_count = sum(c1 != c2 for c1, c2 in zip(domain1, domain2))
            return diff_count == 1  # Only one character different
        return False
    
    def check_spf(self, msg):
        """Check SPF authentication results"""
        auth_results = msg.get('Authentication-Results', '')
        if 'spf=pass' in auth_results:
            return 'pass'
        elif 'spf=fail' in auth_results:
            return 'fail'
        return 'unknown'
    
    def check_dkim(self, msg):
        """Check DKIM signature verification"""
        auth_results = msg.get('Authentication-Results', '')
        if 'dkim=pass' in auth_results:
            return 'pass'
        elif 'dkim=fail' in auth_results:
            return 'fail'
        return 'unknown'
    
    def detect_suspicious_headers(self, msg):
        """Detect suspicious header patterns"""
        suspicious = []
        
        # Check for missing standard headers
        if not msg.get('Message-ID'):
            suspicious.append('Missing Message-ID header')
        
        # Check for suspicious user agents
        user_agent = msg.get('User-Agent', '')
        if 'python' in user_agent.lower() or 'curl' in user_agent.lower():
            suspicious.append('Suspicious User-Agent')
        
        return suspicious
    
    def analyze_content(self, email_body):
        """Analyze email content for phishing indicators"""
        body_lower = email_body.lower()
        found_indicators = []
        
        for indicator in self.phishing_indicators:
            if indicator in body_lower:
                found_indicators.append(indicator)
        
        # Count suspicious URLs
        url_pattern = r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+'
        urls = re.findall(url_pattern, email_body)
        
        return {
            'phishing_keywords': found_indicators,
            'url_count': len(urls),
            'urls': urls[:5]  # First 5 URLs for analysis
        }

# Example usage for educational analysis
analyzer = PhishingAnalyzer()
print("Phishing email analysis tool initialized for educational research")

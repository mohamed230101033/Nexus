# Advanced SIEM Correlation Engine (Educational)
import json
import datetime
from collections import defaultdict, deque
import hashlib

class ThreatDetectionEngine:
    def __init__(self):
        self.event_buffer = deque(maxlen=10000)
        self.correlation_rules = []
        self.threat_indicators = defaultdict(int)
        self.suspicious_patterns = {}
        
    def add_correlation_rule(self, rule):
        """Add a new correlation rule for threat detection"""
        self.correlation_rules.append(rule)
    
    def process_event(self, event):
        """Process incoming security event"""
        # Normalize event timestamp
        event['processed_time'] = datetime.datetime.now().isoformat()
        
        # Add to buffer
        self.event_buffer.append(event)
        
        # Enrich event with additional context
        enriched_event = self.enrich_event(event)
        
        # Apply correlation rules
        threats = self.correlate_event(enriched_event)
        
        # Update threat indicators
        self.update_threat_indicators(enriched_event, threats)
        
        return threats
    
    def enrich_event(self, event):
        """Enrich event with additional context"""
        enriched = event.copy()
        
        # Add geolocation info (mock)
        if 'src_ip' in event:
            enriched['src_geo'] = self.get_geolocation(event['src_ip'])
        
        # Add reputation data
        if 'domain' in event:
            enriched['domain_reputation'] = self.check_domain_reputation(event['domain'])
        
        # Calculate event hash for deduplication
        event_str = json.dumps(event, sort_keys=True)
        enriched['event_hash'] = hashlib.md5(event_str.encode()).hexdigest()
        
        return enriched
    
    def correlate_event(self, event):
        """Apply correlation rules to detect threats"""
        detected_threats = []
        
        for rule in self.correlation_rules:
            if self.evaluate_rule(rule, event):
                threat = {
                    'rule_id': rule['id'],
                    'severity': rule['severity'],
                    'description': rule['description'],
                    'event': event,
                    'timestamp': datetime.datetime.now().isoformat()
                }
                detected_threats.append(threat)
        
        return detected_threats
    
    def evaluate_rule(self, rule, event):
        """Evaluate if an event matches a correlation rule"""
        try:
            # Simple rule evaluation (educational example)
            conditions = rule.get('conditions', [])
            
            for condition in conditions:
                field = condition['field']
                operator = condition['operator']
                value = condition['value']
                
                if field not in event:
                    return False
                
                event_value = event[field]
                
                if operator == 'equals' and event_value != value:
                    return False

            return True
            
        except Exception as e:
            print(f"Rule evaluation error: {e}")
            return False
    
    def get_geolocation(self, ip):
        """Mock geolocation lookup"""
        # In real implementation, use GeoIP database
        if ip.startswith('10.') or ip.startswith('192.168.'):
            return {'country': 'Internal', 'risk': 'low'}
        else:
            return {'country': 'External', 'risk': 'medium'}
    
    def check_domain_reputation(self, domain):
        """Mock domain reputation check"""
        suspicious_domains = ['malware.example', 'phish.test', 'bad.domain']
        if domain in suspicious_domains:
            return {'score': 0, 'category': 'malicious'}
        return {'score': 80, 'category': 'clean'}
    
    def update_threat_indicators(self, event, threats):
        """Update threat indicators based on detected threats"""
        for threat in threats:
            indicator_key = f"{threat['rule_id']}_{event.get('src_ip', 'unknown')}"
            self.threat_indicators[indicator_key] += 1
    
    def get_threat_summary(self):
        """Get summary of current threat landscape"""
        recent_events = list(self.event_buffer)[-100:]  # Last 100 events
        
        return {
            'total_events': len(self.event_buffer),
            'recent_events': len(recent_events),
            'threat_indicators': dict(self.threat_indicators),
            'top_threats': self.get_top_threats()
        }
    
    def get_top_threats(self):
        """Get top threats by frequency"""
        sorted_threats = sorted(
            self.threat_indicators.items(), 
            key=lambda x: x[1], 
            reverse=True
        )
        return sorted_threats[:10]
